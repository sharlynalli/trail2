public class Switch
{  
public static void main(String[] args)
{  
    
    int a =14;  
    
    switch(a)
    {  
     
    case 2: System.out.println("10");  
    break;  
    case 6: System.out.println("13");  
    break;  
    case 30: System.out.println("30");  
    break;  
      
    default:System.out.println("Not in found");  
    }  
}  
}
